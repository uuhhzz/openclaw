main.py 
import openclaw

# 创建 AI 助手
assistant = openclaw.Assistant(model="gpt-4")

# 生成代码
code = assistant.generate(
    prompt="创建一个 Python Web 服务器"
)

# 调试代码
assistant.debug(code)
